CSS

Center an element horizontally and vertically​

.center-element {
  position: absolute;
  top: 50%;
  left: 50%;
  transform: translate(-50%, -50%);
}

Center an element vertically​

.center-vertically {
  position: absolute;
  top: 50%;
  transform: translateY(-50%);
}

Center an element horizontally​

.center-horizontally {
  margin: 0 auto;
}

Create a responsive flexbox container​

.container {
  display: flex;
  flex-wrap: wrap;
  justify-content: space-between;
}

Style buttons​

.button {
  padding: 10px 20px;
  background-color: #007bff;
  color: #fff;
  border: none;
  border-radius: 5px;
  cursor: pointer;
}
.button:hover {
  background-color: #0056b3;
}

Create a gradient background​

.gradient-bg {
  background: linear-gradient(to bottom, #ff9900, #ff5500);
}

Add box shadow​

.box-shadow {

}

.animate-on-hover:hover {

}

Create a pulse animation​

.pulse-animation {
  animation: pulse 1s infinite;
}
@keyframes pulse {
  0% {
    transform: scale(1);
  }
  50% {
    transform: scale(1.1);
  }
  100% {
    transform: scale(1);
  }
}

Style a tooltip​

.tooltip {
  position: relative;
  cursor: pointer;
}
.tooltip-text {
  display: none;
  position: absolute;
  background-color: #333;
  color: #fff;
  padding: 5px;
  border-radius: 3px;
  bottom: 20px;
  left: 50%;
  transform: translateX(-50%);
}
.tooltip:hover .tooltip-text {
  display: block;
}

Create a rotating spinner​

.spinner {
  animation: spin 1s infinite linear;
}
@keyframes spin {
  0% {
    transform: rotate(0deg);
  }
  100% {
    transform: rotate(360deg);
  }
}

Style a table​

.table {

.table th,
.table td {

  padding: 10px;
}

Create a 3D button effect​

.button-3d {
  position: relative;
  background-color: #007bff;
  color: #fff;
  padding: 10px 20px;
}
.button-3d::before {
  content: "";
  position: absolute;
  top: 5px;
  left: 5px;
  right: 5px;
  bottom: 5px;
  background-color: #0056b3;
  z-index: -1;
}

.zoom-in-animation {

  0% {
    transform: scale(0);
  }

    transform: scale(1);
  }
